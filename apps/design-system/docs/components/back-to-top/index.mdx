usage:
  when_to_use:
    - value: Pages with static content exceeding two screens on large/medium displays
    - value: Content longer than four screens on small screens
  when_to_consider_else:
    - value: Pages with brief, short-length static content
    - value: Single-screen pages where scrolling is minimal
  behavior:
    - value: Reveals as a sticky element in the lower right during scrolling
    - value: Displays "Back to top" label with upward arrow on desktop
    - value: Shows arrow icon only on small screens
    - value: Smoothly scrolls to page top when clicked
    - value: Remains fixed at bottom right when user reaches page end
accessibility:
  - value: Animations can be distracting to users with vestibular conditions
  - value: Respects user smooth-scrolling preferences via `prefers-reduced-motion`
  - value: Should programmatically focus near-page-top element (like H1) after click
  - value: Provides clear visible focus state for keyboard users

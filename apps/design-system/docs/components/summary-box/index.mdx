usage:
  when_to_use:
    - value: Highlight the most important information on a page
    - value: Present evergreen content (not time-sensitive)
    - value: Display key details readers shouldn't miss, such as eligibility criteria or coverage details under VA benefits
    - value: Position near the top of the page after the main heading and introductory text
  when_to_consider_else:
    - value: Basic overview information that doesn't require special emphasis
    - value: Related content linking to other pages (use [Link - Collection](/docs/components/link/collection) or [Cards](/docs/components/card) instead)
    - value: Table of contents functionality (use On this page component)
    - value: Dynamic content added via user interaction (use [Alert](/docs/components/alert/) component)
    - value: Time-sensitive messaging (use [Alert](/docs/components/alert/) component)
    - value: Multiple sibling content chunks (use [Accordions](/docs/components/accordion))
    - value: Additional contextual information in forms (use [Additional Info](/docs/components/additional-info) component)
content_considerations:
  - value: Keep content brief and concise
  - value: If content exceeds 5 bullet points, contains bullets longer than 20 words, or requires headers/images/buttons, use standard body text, headings, and lists instead
  - value: Supports heading levels h2-h4, paragraphs, lists, emphasis, and links
accessibility:
  - value: Use appropriate heading levels within the summary box
  - value: Ensure sufficient color contrast for all text
  - value: Content should be readable by screen readers in a logical order

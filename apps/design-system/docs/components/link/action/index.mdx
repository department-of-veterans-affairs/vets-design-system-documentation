usage:
  when_to_use:
    - value: Entice users to start digital services or applications
    - value: Create emphasis on navigational links needing visual prominence
    - value: Direct users to begin a benefit application or online tool
    - value: Display multiple action links of equal importance (secondary variation)
    - value: Provide contrast on dark-background areas (reverse variation)
  when_to_consider_else:
    - value: "**Authentication actions:** Use buttons for \"sign up,\" \"submit,\" or \"sign out\""
    - value: "**Form progression:** Use Button - Primary for moving between form steps"
    - value: "**Non-action navigation:** Don't replace standard links with action links"
content_considerations:
  - value: Keep link text brief and action-oriented
  - value: Start with a verb (e.g., "Apply for health care benefits")
  - value: Ensure purpose and target are clear
accessibility:
  - value: Must indicate they are external
  - value: Default behavior appends "(opens in a new tab)" text

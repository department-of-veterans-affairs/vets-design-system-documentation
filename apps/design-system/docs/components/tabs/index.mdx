  when_to_use:
    - value: Organize related content into separate, focused panels that users switch between
    - value: Reduce cognitive load by presenting information in manageable sections
    - value: Best with 3 or fewer tabs
  when_to_consider_else:
    - value: "**Binary toggles**: Use [Button - Segmented](/docs/components/button/button-segmented) instead (e.g., list vs. map view)"
    - value: "**Sequential/hierarchical content**: Use [Accordion](/docs/components/accordion) or well-formatted single page with [On this page](/docs/components/on-this-page) component"
    - value: "**Navigation**: Use Side Navigation for moving between hierarchy levels"
    - value: "**Search filtering**: Use Search Results pattern"
  behavior:
    - value: First tab selected by default
    - value: Tab panels need 32px spacing above/below
    - value: Horizontal scrollbar appears on overflow (not ideal UX)
    - value: Keep focus on selected tab after activationâ€”don't auto-focus into panel content
content_considerations:
  - value: "Tab labels: 1-2 words, sentence case, plain language"
  - value: Start each panel with a descriptive heading
  - value: Maintain proper heading hierarchy (h2, h3, h4)
  - value: Use clear, descriptive labels (avoid generic labels like "Tab 1")
accessibility:
  - value: Use clear, descriptive labels that indicate panel content
  - value: Place tabs after h1 in main content region
  - value: "Follow W3C standard keyboard navigation patterns:"
  - value: Don't use tabs for page-level navigation
  - value: Ensure proper ARIA roles and attributes are applied

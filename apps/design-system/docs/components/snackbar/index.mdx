usage:
  when_to_use:
    - value: When a user action triggering an API call succeeds or fails
    - value: To inform users about the status of completed actions
    - value: When additional user actions (like retry or undo) are needed in response to the outcome
  when_to_consider_else:
    - value: This component is **mobile app-only** and not intended for web applications
    - value: Not suitable for information that requires user acknowledgment before proceeding
    - value: For web applications, consider using [Alert](/docs/components/alert/) components instead
content_considerations:
  - value: "**Success messages**: Use 2-3 words like \"Draft saved\", \"Appointment canceled\""

  - value: Avoid words like "successful" or "failed"
  - value: No punctuation required
  - value: Keep messages brief and action-oriented
accessibility:
  - value: Must not auto-close; users control dismissal timing
  - value: Screen readers announce snackbar as an Alert immediately upon display
  - value: Each action button is announced as a separate button
  - value: Positioned above bottom navigation to remain accessible and visible
  - value: Color contrast meets accessibility requirements for both success and error states

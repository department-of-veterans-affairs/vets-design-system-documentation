  when_to_use:
    - value: Communicate current state or progress at a glance
    - value: Indicate urgency or required action
    - value: Provide scannable feedback in lists, tables, or dashboards
    - value: Combine color, icons, and short labels for consistent communication
  when_to_consider_else:
    - value: When tags might be confused with buttons
    - value: For labels exceeding 1-3 words
    - value: When the tag functions as a title
    - value: For page-level alerts (use [Alert](/docs/components/alert/) instead)
    - value: For temporary feedback (use dismissible Alert or Snackbar)
    - value: For interactive/clickable elements
content_considerations:

  - value: "**Consistency**: Apply identical language for similar statuses platform-wide"
  - value: "**Semantic Alignment**: Match tag text with color meaning"

accessibility:
  - value: "**Color + Meaning**: Pair color with descriptive text, icons, or context"
  - value: "**Contrast**: All variations meet WCAG 2.2 AA standards"
  - value: "**Keyboard Navigation**: Non-interactive; won't interfere with Tab navigation"
  - value: "**Screen Reader Support**: Announces \"Status\" before content (e.g., \"Status, Approved\")"
  - value: "**Dynamic Content**: Use `aria-live=\"polite\"` for non-urgent updates"
  - value: "**Cognitive Accessibility**: Maintain consistent placement; use simple language"

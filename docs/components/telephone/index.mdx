usage:
  when_to_use:
    - value: Display phone numbers that should be clickable for calling or messaging
    - value: Present contact information with extensions, international codes, or TTY/SMS capabilities
    - value: Make phone numbers accessible on mobile devices
  when_to_consider_else:
    - value: When the text is not an actual phone number meant to be called or messaged
    - value: For displaying numbers that are purely informational
content_considerations:
  - value: "Format phone numbers with hyphens without parentheses: `212-123-1234`"
  - value: Use "+1" prefix only for international audiences
  - value: Extensions should use "ext. 9" format
  - value: Avoid vanity numbers in body copy except for marketing materials
  - value: Include area codes for all phone numbers
accessibility:
  - value: Clickable phone numbers are announced as links by screen readers
  - value: The `messageAriaDescribedby` prop can provide additional context
  - value: TTY numbers should be clearly labeled for users who need them
  - value: Non-clickable numbers should still be formatted consistently

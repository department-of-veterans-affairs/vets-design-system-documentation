usage:
  when_to_use:

    - value: Create sub-navigation for closely related pages
    - value: Organize grouped content requiring secondary navigation
    - value: Pages that are part of a larger section with 5+ related pages
  when_to_consider_else:
    - value: Fewer than 5 pages; may not need full navigation structure
    - value: Navigation within a single page; use [On this page](../on-this-page) component instead
    - value: Links that aren't closely related or in the same website section
    - value: Site-wide navigation; use [Header](../header/) instead
content_considerations:
  - value: Keep navigation labels concise but descriptive
  - value: Use consistent naming conventions across related pages
  - value: Organize items in logical order (by importance, alphabetically, or workflow)
  - value: Limit nesting to two levels maximum for usability
